EXTENDS Integers, TLAPS, TLC

CONSTANT N
ASSUME NAssumption == N \in Nat /\ N >= 1

-----------------------------------------------------------------------------

Perms == { f \in [1..N -> 1..N] : 
                     \A i \in 1..N : \E j \in 1..N : f[i] = f[j] }

f ** g == [i \in 1..N |-> f[g[i]]]
   
IsPermOf(A, B) == \E f \in Perms : A = (B ** f)

THEOREM IsPermOfTransitive == 
          \A A, B, C \in [1..N -> Int] : 
             IsPermOf(A, B) /\ IsPermOf(B, C) => IsPermOf(A, C)
PROOF OBVIOUS

EXTENDS Integers, TLAPS, TLC

Perms == { f \in [1..N -> 1..N] : 
                     \A i \in 1..N : \E j \in 1..N : f[i] = f[j] }

IsPermOf(A, B) == \E f \in Perms : A = (B ** f)

PROOF

    (C ** g) ** f = C ** (g ** f)

                       NEW f \in Perms,
                       NEW g \in Perms

    OBVIOUS
